var div = document.querySelector("#example_2"),
    p = document.querySelector("p"),
    str = '';

div.addEventListener("transitionend", function (event) {
    str = [
        'propertyName: ', event.propertyName, '<br>',
        'elapsedTime: ', event.elapsedTime, '<br>',
        'pseudoElement: ', event.pseudoElement, '<br>',
        '----------------<br>'].join('');
    p.innerHTML += str;
}, true);

document.querySelector("#expand").addEventListener("click", function (e) {
    div.style.width = '150px';
    div.style.height = '100px';
    str = '';
    p.innerHTML = '';
}, true);

document.querySelector("#shrink").addEventListener("click", function (e) {
    div.style.width = '50px';
    div.style.height = '50px';
    str = '';
    p.innerHTML = '';
}, true);
